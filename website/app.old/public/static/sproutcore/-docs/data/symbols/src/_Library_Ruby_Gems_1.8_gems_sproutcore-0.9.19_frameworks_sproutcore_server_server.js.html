 11 // The Server object knows how to send requests to the server and how to
 12 // get things back from the server.  It automatically handles situations
 13 // such as 304 caching and queuing requests to send to the server later if
 14 // the computer becomes disconnected from the internet.
 15 
 16 // The Server object is designed to work with a resource oriented application.
 17 // That is, you do someting like this:
 18 //
 19 // Server.request('resource','verb',{ parameters })
 20 // or
 21 // Server.create('resource',{ parameters })
 22 // Server.refresh('resource',{ parameters })
 23 // Server.update('resource',{ parameters })
 24 // Server.destroy('resource',{ parameters })
 25 //
 26 // parameters include:
 27 // onSuccess -- passes back returned text
 28 // onFailure --
 29 //
 30 //
 31 // SC.Record.refresh
 32 // SC.Record.commit --> create/update
 33 // SC.Record.destroy

 52     if ((!clientData) || (clientData.size == 0)) return ;

 56   // This is the root method for accessing a server resource.  Pass in the
 57   // resource URL, verb name, and any parameters.  There are several special-
 58   // purpose parameters used also:
 59   //
 60   // onSuccess -- function invoked when request completes. Expects the format
 61   //              didSucceed(status,ajaxRequest,cacheCode,context)
 62   // onFailure -- function invoked when request fails. Same format.
 63   // requestContext -- simply passed back.
 64   // cacheCode -- String indicating the time of the last refresh.
 65   // url -- override the default url building with this url.
 66   //
 67   request: function(resource, action, ids, params, method) {
 68 
 69     // Get Settings and Options
 70     if (!params) params = {} ;
 71     var opts = {} ;
 72     var onSuccess = params.onSuccess; delete params.onSuccess;
 73     var onNotModified = params.onNotModified; delete params.onNotModified ;
 74     var onFailure = params.onFailure ; delete params.onFailure ;
 75     var context = params.requestContext ; delete params.requestContext ;
 76     var accept = params.accept ; delete params.accept ;
 77     var cacheCode = params.cacheCode; delete params.cacheCode ;
 78     var url = params.url; delete params.url;
 79 
 80     opts.requestHeaders = {'Accept': 'application/json, text/javascript, application/xml, text/xml, text/html, */*'}
 81     if (accept) opts.requestHeaders['Accept'] = accept ;
 82     if (cacheCode) opts.requestHeaders['Sproutit-Cache'] = cacheCode ;
 83     opts.method = method || 'get' ;
 84 
 85     if (!url) url = this.urlFor(resource, action, ids, params, opts.method) ;
 86 
 87     // handle ids
 88     if (ids && ids.length > 1) {
 89       params.ids = [ids].flatten().join(',') ;
 90     }    
 91     
 92     // adds a custom HTTP header for remote requests
 93     opts.requestHeaders = {'X-SproutCore-Version' : '1.0'}
 94 
 95     // convert parameters.
 96     var parameters = this._toQueryString(params) ;
 97     if (parameters && parameters.length > 0) opts.parameters = parameters ;
 98     
 99     var request = null ; //will container the ajax request
100     
101     // Save callback functions.
102     opts.onSuccess = function(transport) {
103       var cacheCode = request.getHeader('Last-Modified') ;
104       if ((transport.status == '200') && (transport.responseText == '304 Not Modified')) {
105         if (onNotModified) onNotModified(transport.status, transport, cacheCode,context);
106       } else {
107         if (onSuccess) onSuccess(transport.status, transport, cacheCode,context);
108       }
109     } ;
110     
111     opts.onFailure = function(transport) {
112       var cacheCode = request.getHeader('Last-Modified') ;
113       if (onFailure) onFailure(transport.status, transport, cacheCode,context);
114     } ; 
115     
116     console.log('REQUEST: %@ %@'.fmt(opts.method, url)) ;
117     
118     request = new Ajax.Request(url,opts) ;
119   },

121   /**
122     Generates the URL that is going to be called by this server. Note that you
123     should only return relative URLs. You can only call resources that are on
124     the same domain as where this script was downloaded from.
125 
126     @param {String} resource  the URL where the collection of the resource can be queried
127     @param {String} action    the action that should be performed on the resource
128     @param {Array} ids        array of identifiers of your model instances
129     @param {Array} params     parameters that were passed to the SC.Server#request method
130     @param {String} method    the HTTP method that will be used
131     @returns {String} the URL to use in the request to the backend server
132   **/
133   urlFor: function(resource, action, ids, params, method) {
134     var idPart = (ids && ids.length == 1) ? ids[0] : '';
135     return this.urlFormat.format(resource, action) + idPart;
136   },

144   // This is the method called by a collection to get an updated list of
145   // records.
146   listFor: function(opts) {
147     var recordType = opts.recordType ;
148     var resource = recordType.resourceURL() ;
149 
150     if (!resource) return false ;
151     
152     var order = opts.order || 'id' ;
153     if (!(order instanceof Array)) order = [order] ;
154     order = order.map(function(str){
155       return str.decamelize() ; //rubyify
156     }).join(',') ;
157 
158     params = {} ;
159     if (opts.conditions) {
160       var conditions = this._decamelizeData(opts.conditions) ;
161       for(var key in conditions) {
162         params[key] = conditions[key] ;
163       }
164     }
165     
166     params.requestContext = opts ;
167     params.onSuccess = this._listSuccess.bind(this) ;
168     params.onNotModified = this._listNotModified.bind(this) ;
169     params.onFailure = this._listFailure.bind(this) ;
170     if (opts.cacheCode) params.cacheCode = opts.cacheCode ;
171     if (opts.offset) params.offset = opts.offset;
172     if (opts.limit) params.limit = opts.limit ;
173     if (order) params.order = order ;
174     this.request(resource, this._listForAction, null, params, this._listMethod) ;
175   },
176   
177   _listForAction: 'list',
178   _listMethod: 'get',
179   
180   _listSuccess: function(status, transport, cacheCode, context) {
181     var json = eval('json='+transport.responseText) ;
182     if (!json) { console.log('invalid json!'); return; }
183     
184     // first, build any records passed back
185     if (json.records) {
186       this.refreshRecordsWithData(json.records,context.recordType,cacheCode,false);
187     }
188     
189     // next, convert the list of ids into records.
190     var recs = (json.ids) ? json.ids.map(function(guid) {
191       return SC.Store.getRecordFor(guid,context.recordType) ;
192     }) : [] ;
193     
194     // now invoke callback
195     if (context.callback) context.callback(recs,json.count,cacheCode) ;
196   },
197   
198   _listNotModified: function(status, transport, cacheCode, context) {
199     if (context.callback) context.callback(null,null,null) ;
200   },
201   
202   _listFailure: function(status, transport, cacheCode, records) {
203     console.log('listFailed!') ;
204   },

210   // send the records back to create them. added a special parameter to
211   // the hash for each record, _guid, which will be used onSuccess.
212   createRecords: function(records) { 
213     if (!records || records.length == 0) return ;
214 
215     records = this._recordsByResource(records) ; // sort by resource.
216     for(var resource in records) {
217       if (resource == '*') continue ;
218       
219       var curRecords = records[resource] ;
220 
221       // collect data for records
222       var server = this ; var context = {} ;
223       var data = curRecords.map(function(rec) {
224         var recData = server._decamelizeData(rec.getPropertyData()) ;
225         recData._guid = rec._guid ;
226         context[rec._guid] = rec ;
227         return recData ;
228       }) ;
229 
230       // issue request
231       this.request(resource, this._createAction, null, {
232         requestContext: context, 
233         onSuccess: this._createSuccess.bind(this),
234         onFailure: this._createFailure.bind(this),
235         records: data
236       }, this._createMethod) ;
237     }
238   },

243   // This method is called when a create is successful.  It first goes through
244   // and assigns the primaryKey to each record.
245   _createSuccess: function(status, transport, cacheCode, context) {
246     var json = eval('json='+transport.responseText) ;
247     if (!(json instanceof Array)) json = [json] ;
248     
249     // first go through and assign the primaryKey to each record.
250     if (!context) context = {} ;
251     json.each(function(data) {
252       var guid = data._guid ;
253       var rec = (guid) ? context[guid] : null ;
254       if (rec) {
255         var pk = rec.get('primaryKey') ;
256         var dataKey = (pk == 'guid') ? 'id' : pk.decamelize().toLowerCase().replace(/\-/g,'_') ;
257         rec.set(pk,data[dataKey]) ;
258         rec.set('newRecord',false) ;
259       }
260     }) ;
261     
262     // now this method will work so go do it.
263     this.refreshRecordsWithData(json,context._recordType,cacheCode,true) ;
264   },
265   
266   _createFailure: function(status, transport, cacheCode, records) {
267     console.log('createFailed!') ;
268   },

274   refreshRecords: function(records) {
275     if (!records || records.length == 0) return ;
276 
277     records = this._recordsByResource(records) ; // sort by resource.
278     for(var resource in records) {
279       if (resource == '*') continue ;
280       
281       var curRecords = records[resource] ;
282       
283       // collect resource ids, sort records into hash, and get cacheCode.
284       var cacheCode = null ; var ids = [] ; var context = {} ;
285       var primaryKey = curRecords[0].get('primaryKey') ; // assumes all the same
286       curRecords.each(function(r) {
287         cacheCode = cacheCode || r._cacheCode ;
288         var key = r.get(primaryKey);
289         if (key) { ids.push(key); context[key] = r; }
290       });
291       context._recordType = this._instantiateRecordType(curRecords[0].get('type'), this.prefix, null) ; // default rec type.
292       
293       params = {
294         requestContext: context, 
295         cacheCode: ((cacheCode=='') ? null : cacheCode),
296         onSuccess: this._refreshSuccess.bind(this),
297         onFailure: this._refreshFailure.bind(this)
298       };
299       
300       if (ids.length == 1 && curRecords[0].refreshURL) params['url'] = curRecords[0].refreshURL;
301       
302       // issue request
303       this.request(resource, this._refreshAction, ids, params, this._refreshMethod) ;
304     }
305   },

323   // COMMIT
324   
325   commitRecords: function(records) {
326     if (!records || records.length == 0) return ;
327 
328     records = this._recordsByResource(records) ; // sort by resource.
329     for(var resource in records) {
330       if (resource == '*') continue ;
331       
332       var curRecords = records[resource] ;
333 
334       // collect data for records
335       var server = this ;
336 
337       // start format differences
338       var data = null;
339       switch(this.get('postFormat')){
340         case SC.URL_ENCODED_FORMAT:     				
341           data = curRecords.map(function(rec) {
342             return server._decamelizeData(rec.getPropertyData()) ;
343           }) ;
344           break;
345         case SC.JSON_FORMAT:
346           // get all records and put them into an array
347           var objects = [];
348           for(rec in curRecords){
349             if (!curRecords.hasOwnProperty(rec)) continue ;
350             objects.push(curRecords[rec].get('attributes') || {});
351           }
352           
353           // convert to JSON and escape if this.escapeJSON is true
354           if(this.get('escapeJSON')){
355             data = escape(objects.toJSONString());
356           } else {
357             data = objects.toJSONString();
358           }
359           break;
360         default: 
361           break;
362       }
363       // end format differences
364 
365       if (data) {
366         var ids = [];
367         if (curRecords.length == 1) {
368           var primaryKey = curRecords[0].get('primaryKey') ;
369           var key = curRecords[0].get(primaryKey);
370           if (key) ids.push(key);
371         }
372 
373         params = {
374           requestContext: records,
375           onSuccess: this._commitSuccess.bind(this),
376           onFailure: this._commitFailure.bind(this),
377           records: data
378         };
379 
380         if (ids.length == 1 && curRecords[0].updateURL) params['url'] = curRecords[0].updateURL;
381 
382         // issue request
383         this.request(resource, this._commitAction, ids, params, this._commitMethod) ;
384       }
385     }
386   },
387

389   _commitMethod: 'post',

406   destroyRecords: function(records) {
407     if (!records || records.length == 0) return ;
408 
409     records = this._recordsByResource(records) ; // sort by resource.
410     for(var resource in records) {
411       var curRecords = records[resource] ;
412 
413       if (resource == '*') {
414         curRecords.each(function(rec){
415           rec.set('isDeleted',true) ;
416           SC.Store.removeRecord(rec) ;
417         });
418         continue ;
419       }
420 
421       // collect resource ids, sort records into hash, and get cacheCode.
422       var ids = [] ; var key ;
423       var primaryKey = curRecords[0].get('primaryKey') ;
424 
425       curRecords.each(function(rec) {
426         if ((key = rec.get(primaryKey)) && (!rec.get('newRecord'))) {
427           ids.push(key) ; 
428         }
429         rec.set('isDeleted',true) ;
430         SC.Store.removeRecord(rec) ;        
431       }) ;
432 
433       // issue request -- we may not have ids to send tho (for ex, if all
434       // records were newRecords.)
435       if (ids && ids.length > 0) {
436         params = {
437           requestContext: records,
438           onSuccess: this._destroySuccess.bind(this),
439           onFailure: this._destroyFailure.bind(this)
440         };
441 
442         if (ids.length == 1 && curRecords[0].destroyURL) params['url'] = curRecords[0].destroyURL;
443 
444         this.request(resource, this._destroyAction, ids, params, this._destroyMethod) ;
445       }
446     }
447   },

463   // This method is called by the various handlers once they have extracted
464   // their data.
465   refreshRecordsWithData: function(dataAry,recordType,cacheCode,loaded) {
466     var server = this ;
467     
468     // Loop through the data Array and prepare each element
469     var prepedDataAry = [];
470     for (var idx = 0; idx < dataAry.length; idx++)
471     {
472       var currElem = server._prepareDataForRecords(dataAry[idx], server, recordType);
473       if (currElem !== null) prepedDataAry.push(currElem);
474     }
475     
476     // now update.
477     SC.Store.updateRecords(prepedDataAry,server,recordType,loaded) ;
478   },

483   _prepareDataForRecords: function(data, server, defaultType) {
484     if (data === null) {
485         return null;
486     } else if ($type(data) == T_ARRAY) {
487       var that = this;
488       return data.map( function(d) {
489         return that._prepareDataForRecords(d, server, defaultType) ;
490       }) ;
491     } else if ($type(data) == T_HASH) { 
492       data = server._camelizeData(data) ; // camelize the keys received back.
493       if (data.id) {
494         // convert the 'id' property to 'guid'
495         data.guid = data.id;
496         delete data.id;
497       }
498       data.recordType = server._instantiateRecordType(data.type, server.prefix, defaultType);
499       if (data.recordType) {
500         return data;
501       } else {
502         console.log("Data RecordType could not be instantiated!: "+data.type) ;
503         return null; // could not process.
504       }
505     } else {
506       console.log("Unknown data type in SC.Server#_prepareDataForRecords. Should be array or hash.") ;
507       return null; // could not process.
508     }
509   },
510   
511   _instantiateRecordType: function(recordType, prefix, defaultType) {
512     if (recordType) {
513       var recordName = recordType.capitalize() ;
514       if (prefix) {
515         for (var prefixLoc = 0; prefixLoc < prefix.length; prefixLoc++) {
516           var prefixParts = prefix[prefixLoc].split('.');
517           var namespace = window;
518           for (var prefixPartsLoc = 0; prefixPartsLoc < prefixParts.length; prefixPartsLoc++) {
519             var namespace = namespace[prefixParts[prefixPartsLoc]] ;
520           }
521           if (namespace !== window) return namespace[recordName] ;
522         }
523       } else return window[recordName] ;
524     } else return defaultType; 
525   },
526   
527   // places records from array into hash, sorted by resourceURL.
528   _recordsByResource: function(records) {
529     var ret = {} ;
530     records.each(function(rec) {
531       var recs = ret[rec.resourceURL || '*'] || [] ;
532       recs.push(rec)  ;
533       ret[rec.resourceURL || '*'] = recs ;
534     }) ;
535     return ret ;
536   },
537   
538   _camelizeData: function(data) {
539     if (data == null) return data ;
540     
541     // handle array
542     var that = this ;
543     if (data instanceof Array) return data.map(function(d){
544       return that._camelizeData(d) ;
545     }) ;
546 
547     // handle other objects
548     if (typeof(data) == "object") {      
549       var ret = {} ;
550       for(var key in data) {
551         var value = that._camelizeData(data[key]) ;
552         if (key == 'id') key = 'guid' ; 
553         ret[key.replace(/_/g,'-').camelize()] = value ;
554       }
555       return ret ;    
556     }
557     
558     // otherwise just return value
559     return data ;
560   },
561   
562   _decamelizeData: function(data) {
563     if (data == null) return data ;
564     
565     // handle array
566     var that = this ;
567     if (data instanceof Array) return data.map(function(d){
568       return that._decamelizeData(d) ;
569     }) ;
570 
571     // handle other objects
572     if (typeof(data) == "object") {      
573       var ret = {} ;
574       for(var key in data) {
575         var value = that._decamelizeData(data[key]) ;
576         if (key == 'guid') key = 'id' ; 
577         ret[key.decamelize()] = value ;
578       }
579       return ret ;    
580     }
581     
582     // otherwise just return value
583     return data ;
584   },

586   // converts a string, array, or hash into a query string.  root is the 
587   // root string applied to each element key.  Used for nesting.
588   _toQueryString: function(params,rootKey) {
589 
590     // handle nulls
591     if (params == null) {
592       return rootKey + '=';
593       
594     // handle arrays
595     } else if (params instanceof Array) {
596       var ret = [] ;
597       for(var loc=0;loc<params.length;loc++) {
598         var key = (rootKey) ? (rootKey + '['+loc+']') : loc ;
599         ret.push(this._toQueryString(params[loc],key)) ;
600       }
601       return ret.join('&') ;
602       
603     // handle objects
604     } else if (typeof(params) == "object") {
605       var ret = [];
606       for(var cur in params) {
607         var key = (rootKey) ? (rootKey + '['+cur+']') : cur ;
608         ret.push(this._toQueryString(params[cur],key)) ;
609       }
610       return ret.join('&') ;
611       
612     // handle other values
613     } else return [rootKey,params].join('=') ;
614   },
