  9 /**
 10   @class
 11   @extends SC.Object
 12   @since   0.9.14
 13 
 14   @author Charles Jolley
 15   
 16   _IMPORTANT: SC.Collection currently has several known performance issues.  It 
 17   is suitable for use in existing code, but it is possible that the API will 
 18   change before SproutCore 1.0
 19   
 20   A collection holds a set of records matching the specified conditions. You
 21   can set the data source used to find the objects matching the conditions
 22   to create collections pulled from the local set of objects or pulled 
 23   from the server.
 24   
 25   Collection's automatically update their contents based on the conditions
 26   settings you provide as the state of loaded records change.
 27 */
 28 SC.Collection = SC.Object.extend(
 29 /** @scope SC.Collection.prototype */ {
 30   
 31   // ........................................
 32   // CONFIGURABLE PROPERTIES
 33   //
 34   // Set these properties to control the records that will be in this list.
 35   
 36   /**
 37     Indicates the keys to use to order the records.  If you want the records
 38     ordered by descending order, use a string like 'guid DESC'.
 39     
 40     @type {Array}
 41   */
 42   orderBy: ['guid'],
 43 
 44   /**
 45     Set this to the range of records you are interested in seeing or null if
 46     you want to see all records.  Note that collection actually retrieves
 47     a list of records slightly larger than what you pass here to allow for
 48     members to be added and removed.
 49     
 50     @see SC.Collection.offset
 51     
 52     @type {Number}
 53   */
 54   limit: 0,
 55   
 56   /**
 57     Set this to the offset within the range of records you want to retrieve
 58     this.get('limit') records from.
 59     
 60     @see SC.Collection.limit
 61     
 62     @type {Number}
 63   */
 64   offset: 0,
 65   
 66   /**
 67     Set this to a hash with conditions options. e.g. { active: true }.  If you
 68     don't set this property, then all records of the given type will be
 69     used.
 70     
 71     Note also that if you are matching a property that contains an array, the
 72     condition will match if the key you specify is found in the array.  For 
 73     example  { groups: someGroup } would match any record where someGroup is
 74     included in the groups array.
 75     
 76     @type {Object}
 77   */
 78   conditions: {},
 79   
 80   /**
 81     @property
 82   
 83     This is the actual array of records in the current collection. This
 84     property will change anytime the record members change (but not when
 85     the member record properties change).
 86   
 87     @type {Array}
 88   */
 89   records: function() {
 90     if (this._changedRecords) this._flushChangedRecords() ;
 91     return this._records ;
 92   }.property(),
 93   
 94   /**
 95     @property
 96     
 97     The total count of records matching the conditions settings. The contents
 98     of the records array will be clipped to the range value.
 99     
100     @type {Number}
101   */
102   count: function(key, value) {
103     if (value !== undefined) {
104       this._count = value ;
105     } else if (this._changedRecords) this._flushChangedRecords() ;
106     return this._count || 0 ;
107   }.property(),
108 
109   /**
110     Set to true when the collection is destroyed
111     
112     @type {Boolean}
113   */
114   isDeleted: false, // RO
115   
116   /**
117     Set this to the data source you want to use to get the records.  Use
118     either SC.Store or SC.Server.  If you use SC.Server, your recordType
119     must have a resourceURL property.
120     
121     This should be set when the collection is created and not changed later.
122     
123     @type {SC.Store or SC.Server}
124   */
125   dataSource: SC.Store, // NC
126   
127   /**
128     Set this to the type of record you want in the collection.
129     
130     This should be set when the collection is created and not changed later.
131     
132     @type {SC.Record}
133   */
134   recordType: SC.Record, // NC
135 
136   /**
137     Set to true while a refresh is in progress.
138     
139     @type {Boolean}
140   */
141   isLoading: false, // RO
142   
143   /**
144     Set to true if records have changed in a way that might leave the records out
145     of date.
146     
147     @type {Boolean}
148   */
149   isDirty: false, // RO
150   
151   // ........................................
152   // ACTIONS
153   //
154 
155   /**
156     Call this to force the list to refresh.  The refresh may not happen
157     right away, depending on the dataSource.
158   */
159   refresh: function() {
160     var recordType = this.get('recordType') || SC.Record ;
161     var offset = (this._limit > 0) ? this._offset : 0 ;
162     
163     if (!this._boundRefreshFunc) {
164       this._boundRefreshFunc = this._refreshDidComplete.bind(this) ;
165     }    
166     
167     // start refresh
168     if (!this.dataSource) throw "collection does not have dataSource" ;
169     this.beginPropertyChanges();
170     if (!this.isLoading) this.set('isLoading',true) ;
171     this._refreshing = true ;
172     
173     var order = this.get('orderBy') ;
174     if (order && !(order instanceof Array)) order = [order] ;
175     this.dataSource.listFor({
176       recordType: recordType,
177       offset: offset, 
178       limit: this._limit,
179       conditions: this.get('conditions'), 
180       order: order,
181       callback: this._boundRefreshFunc,
182       cacheCode: this._cacheCode
183     }) ;
184     this.endPropertyChanges() ;
185     return this; 
186   },
187   
188   /**
189     Call this method when you are done with a collection.  This will remove
190     it as an observer to changes in the SC.Store so that it can be reclaimed.
191     isDeleted will also be set to true.
192   */
193   destroy: function() { SC.Store.removeCollection(this); return this; },
194   
195   /**
196     Creates a record and immediately adds it to the store. Any collections which
197     match this record will be notified immediately.
198   */
199   newRecord: function(settings) {
200     if (!settings) settings = {} ;
201     settings.newRecord = true ;
202     settings.dataSource = this.get('dataSource') ;
203     var ret = this.recordType.create(settings);
204     SC.Store.addRecord(ret) ; // this will add the record to the collection.
205     return ret;
206   },
207 
208   /**
209     This method removes a record from the store, destroying it.
210   */
211   removeRecords: function(objects) {
212     objects = $A(arguments).flatten();
213     for (var i=0; i<objects.length; i++) {
214      objects[i].destroy();
215     }
216   },
217   
218   // ........................................
219   // INTERNAL
220   //
221   
222   // this is the real offset, limit, and records used by the collection.
223   // this can be a superset of the actual limits seen by outsiders.
224   _offset: 0, _limit: 0, _records: null, _members: null, _store: null,
225   
226   init: function() {
227     arguments.callee.base.call(this) ;
228     SC.Store.addCollection(this) ; // get notified of changes.
229     this._computeInteralOffsetAndLimit() ;
230   },
231   
232   // This is the callback executed when the data source has found the records
233   // matching the passed parameters.  The count is the total count of records
234   // matching the conditions, ignoring the offset and limit.
235   _refreshDidComplete: function(records,count,cacheCode) {
236     if (cacheCode) this._cacheCode = cacheCode;
237 
238     if (records) {
239       this.beginPropertyChanges() ;
240 
241       // update count
242       if (this.get('count') != count) this.set('count',count) ;
243 
244       // update the record store and reslice.
245       this.propertyWillChange('records') ;
246       records = this._store = records.slice() ;
247       this._reslice() ;
248       this.propertyDidChange('records') ;
249 
250       this.endPropertyChanges();
251     }
252     
253     this._refreshing = false ; // only one refresh at a time.
254   },
255 
256   // called by SC.Store whenever an locally stored record has changed state.
257   // This method simply indicates that the records property has changed and saved
258   // the record for later processing.
259   recordDidChange: function(rec) {
260     if (!rec && !rec._guid) return ; // probably an error, but recover anyway.
261     if (!this._changedRecords) this._changedRecords = {} ;
262     this._changedRecords[rec._guid] = rec ;
263     this.propertyWillChange('records') ;
264     this.propertyDidChange('records') ;
265     this.propertyWillChange('count') ;
266     this.propertyDidChange('count') ;
267   },
268 
269   // this method gets called just before the records property is returned.  If 
270   // there are any changed records queued up, the record changes will be 
271   // integrated.
272   _flushChangedRecords: function() {
273     if (!this._changedRecords) return ; // nothing to do.
274 
275     if (this.dataSource != SC.Store) throw "non-local data source is not supported" ;
276 
277     var current = this._store || [] ;
278     var order = this.get('orderBy') || [this.recordType.primaryKey()] ;
279     if (!(order instanceof Array)) order = [order] ;
280     var conditions = this.get('conditions') ;
281 
282     // get the sorted set of changed records, both sorted and as a hash.
283     var records = [] ;
284     var changed = this._changedRecords ;
285     for(var guid in changed) {
286       if (!changed.hasOwnProperty(guid)) continue ;
287       records.push(changed[guid]) ;
288     }
289     records = records.sort(function(a,b){ return a.compareTo(b,order); }) ;    
290     this._changedRecords = null ;
291 
292     // step through the current set of records.  Interpolate changed records, remove records 
293     // that don't belong.
294     var loc = 0 ;
295     while(loc < current.length)  {
296       var working = current[loc] ;
297       var compareToPrev, compareToNext ;
298       
299       // is this record one of the changed?
300       // if so, then the record may need to be removed from its current position if it is deleted,
301       // no longer belongs to the group, or is out of order.
302       if (changed[working._guid]) {
303 
304         var belongs = (!working.get('isDeleted')) && working.matchConditions(conditions) ;
305         if (belongs) {
306           
307           // comes after prev?
308           if (loc>0) {
309             belongs = (working.compareTo(current[loc-1],order) >= 0) ;
310           }
311           
312           // comes before next?
313           if (belongs && (loc+1 < current.length)) {
314             belongs = (working.compareTo(current[loc+1],order) <= 0) ;
315           }
316         }
317         if (!belongs) { current.splice(loc,1); continue; }
318       }
319       
320       // if we get here, then the current record belongs where it is. next see if any of the changed
321       // records need to be inserted here.
322       var goAgain = true ;
323       while ((records.length > 0) && goAgain) {
324         var rec = records[0] ;
325         if ((rec != working) && !rec.get('isDeleted') && rec.matchConditions(conditions)) {
326           if (rec.compareTo(working,order) < 0) {
327             current.splice(loc,0,rec) ;
328             loc++ ;
329           } else goAgain = false ;
330         }
331         if (goAgain) records.shift() ;
332       }
333       
334       // move on to the next record.
335       loc++ ;
336     }
337 
338     // if we get to the end and there are changed records left to process,
339     // they probably need to be added.
340     while(records.length > 0) {
341       var rec = records.shift() ;
342       if (!rec.get('isDeleted') && rec.matchConditions(conditions)) {
343         current.push(rec) ;
344       }
345     }
346         
347 //    if ((this == window.tphotos) && (current.length != this._count)) debugger ;
348     
349     // update the count as well.
350     this._store = current ;
351     this._count = current.length ;
352     this._reslice() ;
353   },
354   
355   // Anytime the properties used to filter the collection change, reslice if possible
356   // then refresh from the data source.
357   propertyObserver: function(observing,target,key,value) {
358     if (target != this) return ;
359 
360     // update the internal properties then refresh.
361     var needsRefresh = false ; 
362     var nv ;
363     value = this.get(key) ; 
364     switch(key) {
365       case 'offset':
366       case 'limit':
367         var oldOffset = this._offset ;
368         var oldLimit = this._limit ;
369         this._computeInteralOffsetAndLimit() ;
370         if ((this._offset == oldOffset) && (this._limit == oldLimit)) {
371           this.propertyWillChange('records') ;
372           this._reslice();
373           this.propertyDidChange('records') ;
374         } else needsRefresh = true ;
375         break ;
376         
377       case 'conditions':
378       case 'orderBy':
379         needsRefresh = true ;
380         break ;
381         
382       default:
383         break ;
384     }
385 
386     // refresh only once per loop. We don't want to refresh multiple times
387     // if the user makes changes to multiple settings.
388     if (needsRefresh && !this._refreshing) {
389       this._refreshing = true ;
390       this._cacheCode = null ;
391       this.set('isLoading',true) ; 
392       this.invokeLater(this.refresh) ;
393     }
394   },
395   
396   // using the current offset and limit properties, compute the internal offset and limit used for
397   // records.
398   _computeInteralOffsetAndLimit: function() {
399     if (this.dataSource != SC.Store) {
400       var v ;
401       this._offset = ((v=this.get('offset')) > this.MARGIN) ? (v-this.MARGIN) : 0 ;
402       this._limit = ((v=this.get('limit')) > 0) ? (v + this.MARGIN) : 0 ; 
403     } else this._offset = this._limit = 0 ;
404   },
405   
406   // This method will slice the _store records based on the current offset and limit.
407   // Is it used internally and thus does not indicate that the computed records
408   // property has changed.
409   _reslice: function() {
410     var offset = this.get('offset') ; 
411     var limit = this.get('limit') ;
412     if ((offset > 0) || (limit > 0)) {
413       var start = offset - this._offset ;
414       var end = start + ((limit <= 0) ? (this._store || []).length : limit) ;
415       this._records = this._store.slice(start,end) ;
416     } else this._records = this._store ;
417   },
418   
419   MARGIN: 10
420   
421 }) ;
