 11 /**
 12   @namespace
 13   
 14   SproutCore implements a variety of enhancements to the built-in String 
 15   object that make it easy to perform common substitutions and conversions.
 16   
 17   Most of the utility methods defined here mirror those found in Prototype
 18   1.6.
 19   
 20   @since SproutCore 1.0
 21 */
 22 SC.String = {
 23   
 24   // Interpolate string. looks for %@ or %@1; to control the order of params.
 25   /**
 26     Apply formatting options to the string.  This will look for occurrences
 27     of %@ in your string and substitute them with the arguments you pass into
 28     this method.  If you want to control the specific order of replacement, 
 29     you can add a number after the key as well to indicate which argument 
 30     you want to insert.  
 31 
 32     Ordered insertions are most useful when building loc strings where values
 33     you need to insert may appear in different orders.
 34 
 35     h3. Examples
 36     
 37     {{{
 38       "Hello %@ %@".fmt('John', 'Doe') => "Hello John Doe"
 39       "Hello %@2, %@1".fmt('John', 'Doe') => "Hello Doe, John"
 40     }}}
 41     
 42     @param args {Object...} optional arguments
 43     @returns {String} formatted string
 44   */
 45   fmt: function() {
 46     // first, replace any ORDERED replacements.
 47     var args = arguments;
 48     var str = this.gsub(/%@([0-9]+)/, function(m) {
 49       return (args[parseInt(m[1],0)-1] || '').toString(); 
 50     }) ;
 51     
 52     // now, replace any remaining %@ items.  Use this indexOf() method b/c
 53     // it is faster than split().
 54     var ret = [] ;
 55     var idx = -1 ;
 56     var loc = 0 ;
 57     var argIdx = 0;
 58     while((idx = str.indexOf("%@",loc)) >= 0) {
 59      // slice off initial part of string and push into ret. update loc.
 60      ret.push(str.slice(loc,idx)) ;
 61      loc = idx + 2 ; // 2 to skip '%@'.
 62      
 63      // add in replacement.
 64      var value = args[argIdx++] ;
 65      if (value && value.toString) value = value.toString() ;
 66      ret.push(value) ;
 67     }
 68     
 69     // include any remaining bits of the string.
 70     if (loc < str.length) {
 71       ret.push(str.slice(loc,str.length)) ;
 72     }
 73     
 74     // join return value.
 75     return (ret.length > 1) ? ret.join('') : ret[0] ;
 76   },
 77 
 78   /**
 79     Localizes the string.  This will look up the reciever string as a key 
 80     in the current Strings hash.  If the key matches, the loc'd value will be
 81     used.  The resulting string will also be passed through fmt() to insert
 82     any variables.
 83     
 84     @param args {Object...} optional arguments to interpolate also
 85     @returns {String} the localized and formatted string.
 86   */
 87   loc: function() {
 88     // NB: This could be implemented as a wrapper to locWithDefault() but
 89     // it would add some overhead to deal with the arguments and adds stack
 90     // frames, so we are keeping the implementation separate.
 91     
 92     var kit = String[String.currentLanguage()];
 93     var str = kit[this] ;
 94     if (!str) str = String.English[this] || this ;
 95     return str.fmt.apply(str,arguments) ;
 96   },
 97 
 98   /**
 99     Works just like loc() except that it will return the passed default 
100     string if a matching key is not found.
101     
102     @param def {String} the default to return
103     @param args {Object...} optional formatting arguments
104     @returns {String} localized and formatted string
105   */
106   locWithDefault: function(def) {
107     var kit = String[String.currentLanguage()];
108     var str = kit[this] ;
109     if (!str) str = String.English[this] || def ;
110     var args = SC.$A(arguments) ;
111     args.shift() ; // escape def.
112     return str.fmt.apply(str, args) ;
113   },
114   
115   /** 
116     Capitalizes a string.
117     @return {String} capitalized string
118   */
119   capitalize: function() {
120     return this.charAt(0).toUpperCase() + this.substring(1) ;
121   },
122 
123   /**
124     Capitalized every word in the string, separated by spaces or dashes.
125     @return {String} titleized string.
126   */
127   titleize: function() {
128     return this.replace(SC.STRING_TITLEIZE_REGEXP, 
129       function(str,separater,character) { 
130         return (character) ? (' ' + character.toUpperCase()) : ' ';
131       }).capitalize() ;
132   },
133   
134   /**
135     Camelizes a string.  This will take any words separated by spaces, dashes
136     or underscores and convert them into camelCase.
137     
138     h3. Examples
139     
140     {{{
141       "my favorite items".camelize() => "myFavoriteItems"
142       "css-class-name".camelize() => "cssClassName"  
143       "action_name".camelize() => "actionName" 
144     }}}
145     
146     @returns {String} camelized string
147   */
148   camelize: function() {
149     var parts = this.split('-'), len = parts.length;
150     if (len == 1) return parts[0];
151 
152     var camelized = this.charAt(0) == '-'
153       ? parts[0].charAt(0).toUpperCase() + parts[0].substring(1)
154       : parts[0];
155 
156     for (var i = 1; i < len; i++)
157       camelized += parts[i].charAt(0).toUpperCase() + parts[i].substring(1);
158 
159     return camelized;
160   },
161 
162   
163   /**
164     Converts the string into a class name.  This method will camelize your 
165     string and then capitalize the first letter.
166   */
167   classify: function() {
168     return this.camelize().capitalize() ;
169   },
170   
171   /**
172     Converts a camelized string into all lower case separated by underscores.
173     
174     @returns {String} the decamelized string.
175   */
176   decamelize: function() { 
177     return this.replace(/([a-z])([A-Z])/g,'$1_$2').toLowerCase();
178   },
179 
180   /**
181     Converts a camelized string or a string with spaces or underscores into
182     a string with components separated by dashes.
183     
184     @returns {String} the dasherized string.
185   */
186   dasherize: function() {
187     return this.decamelize().replace(/[ _]/g,'-') ;  
188   },
189   
190   /**
191     Converts a camelized string or a string with dashes or underscores into
192     a string with components separated by spaces.
193     
194     @returns {String} the humanized string.
195   */
196   humanize: function() {
197     return this.decamelize().replace(/[-_]/g,' ') ;
198   },
199   
200   /**
201     Removes any extra whitespace from the edges of the strings.
202     
203     This method is also aliased as strip().
204     
205     @returns {String} the trimmed string
206   */
207   trim: function () {
208     return this.replace(/^\s+|\s+$/g,"");
209   },
210 
211   /**
212     Explodes a string into an array of characters.
213     
214     @returns {Array}
215   */
216   toArray: function() {
217     return this.split('');
218   }
219   
220 } ;

229 // Add strings for various languages to this collection.  String.loc()
230 // method will try to localize the string passed using the current language.
231 // if the language is not available, it will use English.
232 Object.extend(String,
233 /** @scope String @static */ {
234 
235   /**
236     The current browser language as a two letter code.
237   */
238   browserLanguage: ((navigator.language || navigator.browserLanguage).split('-', 1)[0]),
239   
240   /**
241     If YES, localization will favor the detected language instead of the
242     preferred one.
243   */
244   useAutodetectedLanguage: NO,
245   
246   /**
247     This property is set by the build tools to the current build language.
248   */
249   preferredLanguage: null,
250   
251   /**
252     Returns the hash key to use for loc strings.  The default implementation
253     will autodetect the browser language and look for a loc string to 
254     match.  If it can't find one then it will introspect to find loc strings
255     that are defined and use those instead.
256   */
257   currentLanguage: function () {
258     var ret = (this.useAutodetectedLanguage) ? (this.browserLanguage || this.preferredLanguage || 'en') : (this.preferredLanguage || this.browserLanguage || 'en') ;
259 
260     // then try a couple of normalized forms...
261     if (!this[ret]) ret = this.normalizedLanguage(ret);
262     return ret ;
263   },
264   
265   /**
266     Returns a normalized language string for the two letter country code.
267   */
268   normalizedLanguage: function(ret) {
269     switch(ret) {
270       case 'fr':
271         ret = 'French'; 
272         break ;
273       case 'de':
274         ret = 'German'; 
275         break ;
276       case 'ja':
277       case 'jp':
278         ret = 'Japanese'; 
279         break ;
280       case 'en':
281         ret = 'English' ;
282         break ;
283       
284       case 'es':
285         ret = 'Spanish' ;
286         break;
287         
288       default:
289         ret = "English";
290         break ;
291     }
292     return ret;
293   },
294   
295   /**
296     Adds loc strings for the named language.  This method takes care of 
297     creating the localized string hash if it does not already exist.
298     The language can be one of the following or any two-letter country code.
299     
300     English, French, German, Japanese, Spanish
301     
302     @param language {String} the language code
303     @param strings {Hash} hash of loc strings.
304     @returns {this}
305   */
306   addStringsFor: function(language, strings) {    
307     // convert language to a normalized name...
308     language = String.normalizedLanguage(language) ;
309     if (!String[language]) String[language] = {} ;
310     Object.extend(String[language], strings || {}); 
311     return this;
312   }
313 
314 });
