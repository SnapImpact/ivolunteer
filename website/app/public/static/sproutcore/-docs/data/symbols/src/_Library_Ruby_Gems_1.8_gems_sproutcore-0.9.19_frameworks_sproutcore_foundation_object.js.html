182   /** @private
183     This is a way of performing brute-force introspection.  This searches 
184     through all the top-level properties looking for classes.  When it finds
185     one, it saves the class path name.
186   */
187   _findObjectClassNames: function() {
188     
189     if (SC._foundObjectClassNames) return ;
190     SC._foundObjectClassNames = true ;
191     
192     var seen = [] ;
193     var searchObject = function(root, object, levels) {
194       levels-- ;
195 
196       // not the fastest, but safe
197       if (seen.indexOf(object) >= 0) return ;
198       seen.push(object) ;
199 
200       for(var key in object) {
201         if (key == '__scope__') continue ;
202         if (key == '_type') continue ;
203         if (!key.match(/^[A-Z0-9]/)) continue ;
204 
205         var path = (root) ? [root,key].join('.') : key ;
206         var value = object[key] ;
207 
208 
209         switch($type(value)) {
210         case T_CLASS:
211           if (!value._objectClassName) value._objectClassName = path;
212           if (levels>=0) searchObject(path, value, levels) ;
213           break ;
214 
215         case T_OBJECT:
216           if (levels>=0) searchObject(path, value, levels) ;
217           break ;
218 
219         case T_HASH:
220           if (((root != null) || (path=='SC')) && (levels>=0)) searchObject(path, value, levels) ;
221           break ;
222 
223         default:
224           break;
225         }
226       }
227     } ;
228     
229     searchObject(null, window, 2) ;
230     
231     // Internet Explorer doesn's loop over global variables...
232     if ( SC.isIE() ) {
233       searchObject('SC', SC, 2) ; // get names for the SC classes
234       
235       // get names for the model classes, including nested namespaces (untested)
236       for ( var i = 0; i < SC.Server.servers.length; i++ ) {
237         var server = SC.Server.servers[i];
238         if (server.prefix) {
239           for (var prefixLoc = 0; prefixLoc < server.prefix.length; prefixLoc++) {
240             var prefixParts = server.prefix[prefixLoc].split('.');
241             var namespace = window;
242             var namespaceName;
243             for (var prefixPartsLoc = 0; prefixPartsLoc < prefixParts.length; prefixPartsLoc++) {
244               namespace = namespace[prefixParts[prefixPartsLoc]] ;
245               namepaceName = prefixParts[prefixPartsLoc];
246             }
247             searchObject(namepaceName, namespace, 2) ;
248           }
249         }
250       }
251     }

260   tupleForPropertyPath: function(path,root) {

310   _extendAllProps: function(allProperties, base,ext) {
311     var cprops = base._cprops ; var f = Prototype.emptyFunction ;
312     
313     // first, save any concat props.
314     var concats = {} ;
315     if (cprops) for(var cloc=0;cloc<cprops.length;cloc++) {
316       var p = cprops[cloc]; var p1 = base[p]; var p2 = ext[p] ;
317       p1 = (p1 && p2) ? Array.from(p1).concat(p2) : (p1 || p2) ;
318       concats[p] = p1 ;
319     }
320     
321     // deal with observers, bindings, and properties only if they are not 
322     // defined already in the ext.  If they are defined in ext, they will be
323     // concated using the above code.
324     var bindings = (ext._bindings) ? null : (base._bindings || []).slice() ;
325     var observers = (ext._observers) ? null : (base._observers || []).slice();
326     var properties = (ext._properties) ? null : (base._properties || []).slice() ;
327     var outlets = (ext.outlets) ? null : (base.outlets || []).slice() ;
328     
329     // now copy properties, add superclass to func.
330     for(var key in ext) {
331       // avoid copying builtin methods
332       if (!allProperties && !ext.hasOwnProperty(key)) continue ; 
333 
334       // add super to funcs.  Be sure not to set the base of a func to itself
335       // to avoid infinite loops.
336       var value = (concats.hasOwnProperty(key) ? concats[key] : null) || ext[key] ;
337       if (value && (value instanceof Function) && (!value.base)) {
338         if (value != base[key]) value.base = base[key] || f;          
339       }
340       
341       // Possibly add to a bindings.
342       var keyLen = key.length ;
343       if (bindings && (key.slice(keyLen-7,keyLen) == "Binding")) {
344         bindings.push(key) ;
345         
346       // Also add observers, outlets, and properties for functions...
347       } else if (value && (value instanceof Function)) {
348         if (observers && value.propertyPaths) {
349           observers.push(key) ;
350         } else if (properties && value.dependentKeys) {
351           properties.push(key) ;
352         } else if (outlets && value.autoconfiguredOutlet) {
353           outlets.push(key) ;
354         }
355       }
356 
357       // copy property
358       base[key] = value ;
359     }
360 
361     // copy bindings, observers, and properties if defined.
362     if (bindings) base._bindings = bindings;
363     if (observers) base._observers = observers ;
364     if (properties) base._properties = properties ;
365     if (outlets && outlets.length > 0) base.outlets = outlets ;
366 
367     //console.log('bindings: %@ -- observers: %@ -- properties: %@'.format(base._bindings,base._observers,base._properties)) ;
368     
369     return base ;

635   /**  
636     Activates any outlet connections in the the object.  This is called 
637     automatically for views typically.
638     
639     A view may contain outlets.  Outlets are a way to find and connect to
640     elements within the view.
641     
642     @param key {String} optional single key to awake.
643     @returns {void}
644   */
645   awake: function(key) { 
646     // if a key is passed, convert that from an outlet and awake it. otherwise
647     // awake self.
648     if (key !== undefined) {
649       var obj = this.outlet(key) ;
650       if (obj) obj.awake() ;
651       return ;
652     }
653 
654     if (this._awake) return ;
655     this._awake = true ;
656     
657     // it would be cool to do this in a recursive way, but sadly we cannot
658     // without a stack overflow problem. Just loop through outlets and collect
659     // items to awake.
660     this.bindings.invoke('relay') ; 
661     
662     if (this.outlets && this.outlets.length) {
663       var stack = [] ;
664       var working = [this, this.outlets.slice()] ;
665       while(working) {
666         // find the next item to work on.
667         var next = working[1].pop() ;
668         var obj = working[0] ;
669         
670         // an item was found in the array. Process it.
671         if (next) {
672           next = obj[next] ;

674             // awake these bindings.
675             if (next.bindings) next.bindings.invoke('relay') ;

677             // next has outlets itself. Start a new context and process them.
678             if (next.outlets && next.outlets.length > 0) {
679               stack.push(working) ;
680               working = [next,next.outlets.slice()] ;
681             }
682           }
683           
684         // no more items found in the current array. pop the stack.
685         } else working = stack.pop() ;
686       }

780 // CHAIN OBSERVER
781 //
782 
783 // ChainObservers are used to automatically monitor a property several 
784 // layers deep.
785 // org.plan.name = SC._ChainObserver.create({
786 //    target: this, property: 'org',
787 //    next: SC._ChainObserver.create({
788 //      property: 'plan',
789 //      next: SC._ChainObserver.create({
790 //        property: 'name', func: myFunc
791 //      })
792 //    })
793 //  })
794 //
795 SC._ChainObserver = SC.Object.extend({
796   
797   isChainObserver: true,
798   
799   // the object this link in the chain is observing.
800   target: null,
801     
802   // the property on the object this link is observing
803   property: null,
804     
805   // if not null, the next link in the chain.  The target property
806   // of this next link will be set to the value of this link.
807   next: null,
808     
809   // if not null, the function to be called when the property value changes.
810   func: null,
811     
812   propertyObserver: function(observing,target,key,value) {
813     if ((key == 'target') && (value != this._target)) {
814       var func = this.boundObserver() ;
815       if (this._target && this._target.removeObserver) {
816         this._target.removeObserver(this.property,func);
817       } 
818       this._target = value ;
819       if (this._target && this._target.addObserver) {
820         this._target.addObserver(this.property,func) ;
821       }
822       
823       // the target property has probably changed since the target has
824       // changed. notify myself of this.
825       if (!(observing == 'init')) this.targetPropertyObserver() ;
826     }
827   },
828   
829   boundObserver: function() {
830     if (!this._boundObserver) {
831       this._boundObserver = this.targetPropertyObserver.bind(this) ;
832     }
833     return this._boundObserver ;
834   },
835   
836   // invoked when the target property changes.  act based 
837   targetPropertyObserver: function() {
838     
839     // get the new target property value.  if the target or property is 
840     // not valid, just use null.
841     var value = (this.target && this.target.get && this.property) ? this.target.get(this.property) : null ;
842 
843     if (value !== this._lastTargetProperty) {
844       this._lastTargetProperty = value ;
845       // if we have another item in the chain, just forward the change.
846       if (this.next) {
847         this.next.set('target',value) ;
848         
849       // otherwise, invoke the function.
850       } else if (this.func) this.func(this.target,this.property,value) ;
851     }
852     
853   },
854   
855   // hookup observer.
856   init: function() {
857     arguments.callee.base.call(this) ;
858     this.propertyObserver('init',this,'target',this.get('target')) ;
859   }
860     
861 }) ;
