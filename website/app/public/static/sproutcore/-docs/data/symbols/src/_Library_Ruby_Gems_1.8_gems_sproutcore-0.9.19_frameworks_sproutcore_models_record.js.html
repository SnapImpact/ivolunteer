  8 /**
  9   @class
 10 
 11   A Record is the core model class in SproutCore. It is analogous to 
 12   NSManagedObject in Core Data and EOEnterpriseObject in the Enterprise
 13   Objects Framework (aka WebObjects), or ActiveRecord::Base in Rails.
 14   
 15   To create a new model class, in your SproutCore workspace, do:
 16   
 17   {{{
 18     $ sc-gen model my_app/my_model
 19   }}}
 20 
 21   This will create MyApp.MyModel in clients/my_app/models/my_model.js.
 22   
 23   The core attributes hash is used to store the values of a record in a 
 24   format that can be easily passed to/from the server.  The values should 
 25   generally be stored in their raw string form.  References to external 
 26   records should be stored as primary keys.
 27   
 28   Normally you do not need to work with the attributes hash directly.  
 29   Instead you should use get/set on normal record properties.  If the 
 30   property is not defined on the object, then the record will check the 
 31   attributes hash instead.
 32   
 33   You can bulk update attributes from the server using the 
 34   updateAttributes() method.
 35 
 36   @extends SC.Object
 37   @since SproutCore 1.0
 38 */

207   /**
208     Gets an attribute, converting it to the proper format.
209   
210     @param {string} key the attribute you want to read
211     @returns {value} the value of the key, or null if it doesn't exist
212   **/
213   readAttribute: function(key) {
214     if (!this._cachedAttributes) this._cachedAttributes = {} ;
215     var ret = this._cachedAttributes[key] ;
216     if (ret === undefined) {
217       var attr = this._attributes ;
218       ret = (attr) ? attr[key] : undefined ;
219       ret = ret || this[key] ; // also check properties...
220       if (ret !== undefined) {
221         var recordType = this._getRecordType(key+'Type') ;
222         ret = this._propertyFromAttribute(ret, recordType) ;
223       }
224       this._cachedAttributes[key] = ret ;
225     }
226     return (ret === undefined) ? null : ret;
227   },
228 
229   /**
230     Updates the attribute, converting it back to the property format.
231   
232     @param {String} key the attribute you want to read
233     @param {Object} value the attribute you want to read
234     @returns {Object} the value of the key, or null if it doesn't exist
235   **/
236   writeAttribute: function(key, value) {
237     var recordType = this._getRecordType(key+'Type') ;
238     var ret = this._attributeFromProperty(value, recordType) ;
239     if (!this._attributes) this._attributes = {} ;
240     this._attributes[key] = ret ;
241     if (this._cachedAttributes) delete this._cachedAttributes[key];  // clear cache.
242     this.recordDidChange() ;
243     return value ;  
244   },

254   /**
255     This will take the incoming set of attributes and update internal set.
256     
257     Note that if the attributes have never been set, then the object you pass 
258     in may become the new set of attribute.  This assumes the attrs you pass 
259     in will not be modified later.  This method also assumes it is coming from 
260     the server, so the change count will be reset.
261   
262     @param {Object} newAttrs the new attributes for the object
263     @param {Boolean} replace should the overwrite the in-place attributes, or  replace them entirely
264     @returns {Boolean} isLoaded is the object loaded?
265   **/
266   updateAttributes: function(newAttrs, replace, isLoaded) {
267     var changed = false ;
268     if (this._attributes && (replace !== true)) {
269       for(var key in newAttrs) {
270         if (!newAttrs.hasOwnProperty(key)) continue ;
271         if (!changed) changed = (this._attributes[key] != newAttrs[key]) ;
272         this._attributes[key] = newAttrs[key] ;
273       }
274     } else {
275       this._attributes = newAttrs ;
276       changed = true ;
277     }
278     
279     this._cachedAttributes = {} ; // reset cache.
280     
281     if (changed) {
282       this.beginPropertyChanges() ;
283       this.set('changeCount',0) ;
284       this.set('isLoaded',isLoaded) ;
285       this.allPropertiesDidChange() ;
286       this.endPropertyChanges() ;
287 
288       if (SC.Store) SC.Store.recordDidChange(this) ;
289     }

310   unknownProperty: function( key, value )

312     if (value !== undefined) {
313

315       // this record is cached. store the old key, update the value, then let 
316       // the store do the housekeeping...

318       if (key == primaryKeyName)

321         var newPrimaryKey  = value;
322       }
323       
324       this.writeAttribute(key,value);
325       
326       // no need to relocate if there wasn't an old key...
327       if ((key == primaryKeyName) && oldPrimaryKey) SC.Store.relocateRecord( oldPrimaryKey, newPrimaryKey, this );
328       
329     } else {
330       value = this.readAttribute(key);
331     }
332     return value;

343       if (typeConverter) return typeConverter(value,'out') ;

345         return (value) ? value.get(recordType.primaryKey()) : null ;

350   _propertyFromAttribute: function(value,recordType) {
351     if (value && value instanceof Array) {
352       var max = value.get('length') ;
353       var ret = new Array(max) ;

355         var v = value.objectAt(idx) ; 
356         ret[idx] = this._propertyFromAttribute(v, recordType) ; 
357       }

359       return ret ;

361     } else {
362       var typeConverter = this._pickTypeConverter(recordType) ;
363       if (typeConverter) return typeConverter(value,'in') ;
364       if (recordType) {
365         if (!value) return null ;
366         return SC.Store.getRecordFor(value,recordType) ;
367       } else return value ;
368     }

386   /**
387     Compares the receiver to the passed object, using the array of keys to
388     determine the order.  You can use this method as part of a call to sort()
389     on an array.
390     
391     @param object {SC.Record} the other record
392     @param orderBy {Array} array of one or more keys. Optional.
393     @returns {Number} -1, 0, 1
394   */
395   compareTo: function(object, orderBy) {
396     if (!orderBy) orderBy = [this.get('primaryKey')] ;
397     var ret = SC.Record.SORT_SAME ; var loc ;
398     for(loc=0; (ret == SC.Record.SORT_SAME && loc<orderBy.length); loc++) {
399       var key = orderBy[loc] ;
400       
401       // determine order
402       var asc = true ;
403       if (key.match(/ DESC$/)) { 
404         asc = false; key = key.slice(0,-5); 
405       } else if (key.match(/ ASC$/)) {
406         asc = true; key = key.slice(0,-4); 
407       }
408 
409       // if key contains a .dot then we need to get the value for the key.
410       var keys = key.split('.') ;
411       key = keys.shift() ;
412       
413       // get values for key.
414       var a = this.valueForSortKey(key) ;
415       var b = object.valueForSortKey(key) ;
416       
417       // convert the values to comparable values.
418       a = this._comparableValueFor(a,keys) ;
419       b = this._comparableValueFor(b,keys) ;
420       
421       // compare values
422       if (asc) {
423         ret = (a<b) ? SC.Record.SORT_BEFORE : ((a>b) ? SC.Record.SORT_AFTER : SC.Record.SORT_SAME) ;
424       } else {
425         ret = (a>b) ? SC.Record.SORT_BEFORE : ((a<b) ? SC.Record.SORT_AFTER : SC.Record.SORT_SAME) ;
426       }
427     }
428     return ret ;
429   },
430   
431   _comparableValueFor: function(value, keys) {
432     if (keys && keys.length > 0) {
433       var key ; var loc = 0 ;
434       while(value && (loc < keys.length)) {
435         key = keys[loc]; 
436         value = (value.get) ? value.get(key) : value[key] ;
437         loc++ ;
438       }
439     
440     // handle records.
441     } else value = (value && value._guid) ? value._guid : value ;
442     return value ;
443   },

445   /**  
446     Used to match records to a set of conditions.  By default, this will
447     call matchCondition on each condition.
448     
449     @param conditions {Hash} hash of conditions
450     @returns {Boolean} true if the receiver matches the hash of conditions.
451   */
452   matchConditions: function(conditions) {
453     for(var key in conditions) {
454       var value = conditions[key] ;
455       if (value instanceof Array) {
456         var loc = value.length ; var isMatch = false ;
457         while(--loc >= 0) {
458           if (this.matchCondition(key,value[loc])) isMatch = true ;
459         }
460         if (!isMatch) return false ;
461       } else if (!this.matchCondition(key,value)) return false ;      
462     }
463     return true ;
464   },
465 
466   /**
467     Returns true if the value of key matches the passed value.  This is used
468     by matchConditions().
469      
470     @param key {String} the key name
471     @param value {Object} the value to match agains
472     @returns {Boolean} true if matched
473   */
474   matchCondition: function(key, value) {
475     var recValue = this.get(key) ;    
476     var isMatch ;
477     var loc ;
478 
479     // The passed in value appears to be another record instance.
480     // just check for equality with the record as an optimization.
481     if (value && value.primaryKey) { 
482       if ($type(recValue) === T_ARRAY) {
483         loc = recValue.length ;
484         while(--loc >= 0) { 
485           if (recValue === value) return true; 
486         }
487       } else return recValue === value ;
488 
489     // Otherwise, do a more in-depth compare
490     } else { 
491       if ($type(recValue) === T_ARRAY) {
492         loc = recValue.length ;
493         while(--loc >= 0) { 
494           if (this._matchValue(recValue[loc],value)) return true; 
495         }
496       } else return this._matchValue(recValue,value) ;
497     }
498     return false ;
499   },
500 
501   _matchValue: function(recValue,value) {
502     // if we get here with recValue as a record, we must compare by guid, so grab it
503     if (recValue && recValue.primaryKey) recValue = recValue.get(recValue.get('primaryKey')) ;
504     var stringify = (value instanceof RegExp);
505     if (stringify)  {
506       if (recValue == null) return false ;
507       return recValue.toString().match(value) ;
508     } else {
509        return recValue==value ;
510     }
511   },

535   /**
536     This method should be used by the server to push updated data into a
537     record.  The data should be a hash with strings and arrays.  This will
538     use any types you define to convert the values into their correct type.
539     Note that references to external objects should be a string with the
540     primaryKey value of the record.
541   
542     @param data {Hash} the data hash
543     @param isLoaded {Boolean} true if the hash contains a full set of data for the record vs just a summary.
544     @returns {void}
545   */  
546   updateProperties: function(data,isLoaded) {
547     var rec = this ;
548     
549     // for each property, if there is a value in the passed data, convert it to
550     // the configured type.
551     this.beginPropertyChanges() ;
552     if (isLoaded) this.set('isLoaded',true) ;
553     try {
554       var loc = this.properties.length ;
555       while(--loc >= 0) {
556         var prop = this.properties[loc] ;
557         var newValue = data[prop] ;
558 
559         //if (prop == 'tags') debugger ;
560         
561         // handle null values
562         if (newValue === null) {
563           if (rec.get(prop) != null) rec.set(prop,null) ;
564           
565         // handle defined, non-null values
566         } else if (newValue !== undefined) {
567           
568           var oldValue = rec.get(prop) ;            
569           
570           // get type information
571           var recordType = rec.get(prop + 'Type') ;
572           var typeConverter = this._pickTypeConverter(recordType) ;
573           if (typeConverter) recordType = null ;
574  
575           // if array, convert each object.
576           var isSame ; var rec = this ;
577           if (newValue instanceof Array) {
578             newValue = newValue.map(function(nv) {
579               return rec._convertValueIn(nv,typeConverter,recordType) ;
580             }) ;
581             isSame = newValue.isEqual(oldValue) ;
582           } else {
583             newValue = this._convertValueIn(newValue,typeConverter,recordType);
584             isSame = newValue == oldValue ;
585           }
586           
587           // set value
588           if (!isSame) this.set(prop,newValue) ;
589            
590         }
591       }
592     }
593     
594     catch(e) {
595       console.log(this._guid + ': Exception raised on UPDATE: ' + e) ;
596     }
597 
598     this.endPropertyChanges() ;   
599     this.set('changeCount',0) ;        
600   },

602   // this is used for the update.  It should return a hash with current state
603   // of the record.  This uses the types to automatically marshall properties.
604   getPropertyData: function() {
605     var ret = {} ;
606     var properties = this.get('properties') || []; var loc = properties.length;
607     while(--loc >= 0) {
608       var key = properties[loc] ;
609       var value = this.get(key) ;
610       var recordType = this[key + 'Type'] ;
611       var typeConverter = this._pickTypeConverter(recordType) ;
612       if (typeConverter) recordType = null ;
613       
614       // if there is a type, use that to make the conversion.
615       if (value instanceof Array) {
616         var ary = [] ;
617         for(var vloc=0;vloc<value.length;vloc++) {
618           var v = value[vloc] ;
619           ary.push(this._convertValueOut(v,typeConverter,recordType));          
620         }
621         value = ary ;
622       } else value = this._convertValueOut(value,typeConverter,recordType);
623       
624       // set key
625       ret[key] = value ;
626     }
627     return ret ;
628   },
629   
630   _pickTypeConverter: function(recordType) {
631     var typeConverter = null ;
632     if (recordType && recordType.isTypeConverter) {
633       typeConverter = recordType; recordType = null ;
634     } else if(recordType) switch(recordType) {
635       case Date:
636         typeConverter = SC.Record.Date; recordType = null ;
637         break ;
638       case Number:
639         typeConverter = SC.Record.Number; recordType = null;
640         break;
641       case String:
642         typeConverter = null; recordType = null ;
643         break ;
644     }          
645     return typeConverter;
646   },
647   
648   _convertValueOut: function(value,typeConverter,recordType) {
649     if (typeConverter) return typeConverter(value,'out') ;
650     if (recordType) {
651       return (value) ? value.get(recordType.primaryKey()) : null ;
652     } else return value ;
653   },
654   
655   _convertValueIn: function(value,typeConverter,recordType) {
656     if (typeConverter) return typeConverter(value,'in') ;
657     if (recordType) {
658       return SC.Store.getRecordFor(value,recordType) ;
659     } else return value ;
660   },

740   hasMany: function(recordTypeString,conditionKey,opts) {
741     opts = (opts === undefined) ? {} : Object.clone(opts) ;
742     var conditions = opts.conditions || {} ;
743     opts.conditions = conditions ;
744 
745     var privateKey = '_' + conditionKey + SC.generateGuid() ;
746     return function() {
747       if (!this[privateKey]) {
748         var recordType = eval(recordTypeString);
749         conditions[conditionKey] = this ;
750         this[privateKey] = recordType.collection(opts) ;
751         this[privateKey].refresh() ; // get the initial data set.
752       }
753       return this[privateKey] ;
754     }.property();
755   },

766     }
767     rec.endPropertyChanges() ;

776 // Built in Type Converters.  You can also use an SC.Record.
777 SC.Record.Date = function(value,direction) {
778   if (direction == 'out') {
779     if (value instanceof Date) value = value.utcFormat() ;
780     
781   } else if (typeof(value) == "string") {
782     // try to parse date. trim any decimal numbers at end since Rails sends
783     // this sometimes.
784     var ret = Date.parseDate(value.replace(/\.\d+$/,'')) ;
785     if (!ret) ret = new Date(value);
786     if (ret) value = ret ;
787   }
788   return value ;
789 }.typeConverter() ;

791 SC.Record.Number = function(value,direction) {
792   if (direction == 'out') {
793     if (typeof(value) == "number") value = value.toString() ;
794   
795   } else if (typeof(value) == "string") {
796     var ret = (value.match('.')) ? parseFloat(value) : parseInt(value,0) ;
797     if (ret) value = ret ;
798   }
799   return value ;
800 }.typeConverter() ;

802 SC.Record.Flag = function(value, direction) {
803   if (direction == 'out') {
804     return value = (value) ? 't' : 'f' ;
805   } else if (typeof(value) == "string") {
806     return !('false0'.match(value.toLowerCase())) ;
807   } else return (value) ? true : false ;
